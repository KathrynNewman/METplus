148  def run_at_time_loop_string(self, input_dict):
149  """!Read TCPairs track data into TCStat to filter the data. Using the
150  resulting track data, run RegridDataPlane on the model data to
151  create tiles centered on the storm.
152 
153  @param input_dict dictionary containing initialization time
154  """
155 
156  # Calculate other time information from available time info
157  time_info = time_util.ti_calculate(input_dict)
158 
159  self.logger.debug("Begin extract tiles")
160  init_fmt = time_info['init'].strftime('%Y%m%d_%H')
161 
162  # Create the name of the filter file to use
163  filter_filename = (
164  do_string_sub(self.c_dict['TC_STAT_INPUT_TEMPLATE'],
165  **time_info)
166  )
167  filter_path = os.path.join(self.c_dict['TC_STAT_INPUT_DIR'],
168  filter_filename)
169 
170  self.logger.debug(f"Looking for input stat file: {filter_path}")
171  if not os.path.exists(filter_path):
172  self.log_error(f"Could not find input stat file: {filter_path}")
173  return
174 
175  # Now get unique storm ids from the filter file
176  # store list of lines from tcst file for each storm_id as the value
177  storm_dict = util.get_storms(filter_path)
178 
179  if not storm_dict:
180  # No storms found for init time, init_fmt
181  self.logger.debug(f"No storms were found for {init_fmt}"
182  "...continue to next in list")
183  return
184 
185  # get indices of values from header
186  idx_dict = self.get_header_indices(storm_dict['header'])
187 
188  # Create tiles for each storm in the storm_dict dictionary
189  for storm_id, storm_lines in storm_dict.items():
190  if storm_id == 'header':
191  continue
192 
193  self.create_tiles_from_storm(storm_id,
194  storm_lines,
195  idx_dict)
196 
197  util.prune_empty(self.c_dict['OUTPUT_DIR'], self.logger)

199  def create_tiles_from_storm(self, storm_id, storm_lines, idx_dict):
200  """! Run RegridDataPlane to create a tile for forecast and observation
201  data for each storm track point
202 
203  Args:
204  @param storm_id value from STORM_ID column to process
205  @param storm_lines Each line from tcst file for given storm id
206  @param idx_dict dictionary of indices for each header value
207  """
208 
209  # loop over storm track
210  for storm_line in storm_lines:
211  storm_data = self.get_storm_data_from_track_line(idx_dict,
212  storm_line)
213 
214  time_info = self.set_time_info_from_storm_data(storm_id,
215  storm_data)
216 
217  # set var list from config using time info
218  var_list = util.parse_var_list(self.config,
219  time_info,
220  met_tool=self.app_name)
221 
222  # set output grid and run for the forecast and observation data
223  for dtype in ['FCST', 'OBS']:
224  self.regrid_data_plane.c_dict['VERIFICATION_GRID'] = (
225  self.get_grid(dtype, storm_data)
226  )
227 
228  # run RegridDataPlane wrapper
229  ret = self.regrid_data_plane.run_at_time_once(time_info,
230  var_list,
231  data_type=dtype)
232  self.all_commands.extend(self.regrid_data_plane.all_commands)
233  self.regrid_data_plane.all_commands.clear()
234 
235  # if RegridDataPlane failed to run for FCST, skip OBS
236  if not ret:
237  break

325  def get_grid_info(self, lat, lon, data_type):
326  """! Create the grid specification string with the format:
327  latlon Nx Ny lat_ll lon_ll delta_lat delta_lon
328  used by the MET tool, regrid_data_plane.
329 
330  @param lat The latitude of the grid point
331  @param lon The longitude of the grid point
332  @param data_type FCST or OBS, used for log output only
333  @returns the tile grid string for the input lat and lon
334  """
335  nlat = self.c_dict['NLAT']
336  nlon = self.c_dict['NLON']
337  dlat = self.c_dict['DLAT']
338  dlon = self.c_dict['DLON']
339 
340  # Format for regrid_data_plane:
341  # latlon Nx Ny lat_ll lon_ll delta_lat delta_lonadj_lon =
342  # float(lon) - lon_subtr
343  adj_lon = float(lon) - self.c_dict['LON_ADJ']
344  adj_lat = float(lat) - self.c_dict['LAT_ADJ']
345  lon0 = str(util.round_0p5(adj_lon))
346  lat0 = str(util.round_0p5(adj_lat))
347 
348  self.logger.debug(f'{data_type} lat: {lat} => {lat0}, '
349  f'lon: {lon} => {lon0}')
350 
351  grid_def = f"latlon {nlat} {nlon} {lat0} {lon0} {dlat} {dlon}"
352 
353  return grid_def
