467  def get_fcst_file_info(self, output_dir, storm_id):
468  """! Get the number of all the gridded forecast n x m tile
469  files for a given storm id and init time
470  (that were created by extract_tiles). Determine the filename of the
471  first and last files. This information is used to create
472  the title value to the -title opt in plot_data_plane.
473 
474  @param output_dir Directory containing ASCII list file of
475  forecast files to process
476  @param storm_id ID of storm to process
477  @returns num, beg, end: A tuple representing the number of
478  forecast tile files, and the first and last file. If info cannot
479  be parsed, return (None, None, None)
480  """
481  fcst_path = os.path.join(output_dir,
482  f"{self.FCST_ASCII_FILE_PREFIX}{storm_id}")
483  # read the file but skip the first line because it contains 'file_list'
484  with open(fcst_path, 'r') as file_handle:
485  files_of_interest = file_handle.readlines()[1:]
486 
487  # Get a sorted list of the forecast tile files for the init
488  # time of interest for all the storm ids and return the
489  # forecast hour corresponding to the first and last file.
490  sorted_files = sorted(files_of_interest)
491  if not files_of_interest:
492  self.log_error(f"No files found in file list: {fcst_path}")
493  return None, None, None
494 
495  first = sorted_files[0]
496  last = sorted_files[-1]
497 
498  # Extract the forecast hour from the first and last filenames.
499  fcst_regex = f".*{self.c_dict['FCST_TILE_PREFIX']}" + "([0-9]{3}).*.nc"
500  match_beg = re.search(fcst_regex, first)
501  match_end = re.search(fcst_regex, last)
502 
503  if match_beg:
504  beg = f"F{match_beg.group(1)}"
505  else:
506  self.log_error("Unexpected file format encountered, exiting...")
507  return None, None, None
508 
509  if match_end:
510  end = f"F{match_end.group(1)}"
511  else:
512  self.log_error("Unexpected file format encountered, exiting...")
513  return None, None, None
514 
515  # Get the number of forecast tile files
516  num = str(len(sorted_files))
517 
518  return num, beg, end
