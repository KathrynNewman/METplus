285  def compare_time_info(self, runtime, filetime):
286  """! Compare current runtime dictionary to current file time dictionary
287  If runtime value for init, valid, or lead is not a wildcard and
288  it doesn't match the file's time value, return False. Otherwise
289  return True.
290 
291  @param runtime dictionary containing time info for current runtime
292  @param filetime dictionary containing time info for current file
293  @returns True if file's info matches the requirements for current
294  runtime or False if not.
295  """
296  for time_val in ['init', 'valid']:
297  if (runtime[time_val] != '*' and
298  filetime[time_val] != runtime[time_val]):
299  return False
300 
301  if runtime['lead'] == '*':
302  return True
303 
304  # convert each value to seconds to compare
305  runtime_lead = time_util.ti_get_seconds_from_lead(runtime['lead'],
306  runtime['valid'])
307  filetime_lead = time_util.ti_get_seconds_from_lead(filetime['lead'],
308  filetime['valid'])
309  # if cannot compute seconds, possibly using months or years, so compare
310  # forecast leads directly
311  if runtime_lead is None or filetime_lead is None:
312  return runtime['lead'] == filetime['lead']
313 
314  return runtime_lead == filetime_lead
